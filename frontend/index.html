document.getElementById('fleetForm').addEventListener('submit', async function (event) { event.preventDefault(); // Collect input data const licensePlate = document.getElementById('licensePlate').value; const distanceTraveled = parseFloat(document.getElementById('distanceTraveled').value); const loadCapacity = parseFloat(document.getElementById('loadCapacity').value); const truckLifetime = parseInt(document.getElementById('truckLifetime').value); // Send data to backend for processing (replace '/api/calculate' with your server endpoint) const response = await fetch('/api/calculate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ licensePlate, distanceTraveled, loadCapacity, truckLifetime }) }); if (response.ok) { const result = await response.json(); // Display results document.getElementById('savings').textContent = result.savings; document.getElementById('emissionReduction').textContent = result.emissionReduction; document.getElementById('subsidies').textContent = result.subsidies.join(', '); document.getElementById('results').style.display = 'block'; } else { alert('An error occurred while processing your request.'); } });
