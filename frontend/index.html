@app.route('/calculate', methods=['POST']) def calculate(): # Accept data sent to the server as JSON data = request.get_json() # Get data from the request (example fields) distance = data.get('distance') # Distance traveled in km fuel_efficiency = data.get('fuel_efficiency') # Fuel efficiency (liters per km) emission_factor = data.get('emission_factor') # CO2 emissions per liter of fuel (example) # Example calculation: fuel cost fuel_cost_per_liter = 1.5 # Euro per liter of fuel (example value) total_fuel_cost = distance * fuel_efficiency * fuel_cost_per_liter # Example CO2 emissions calculation (kg CO2) total_emissions = distance * fuel_efficiency * emission_factor # Send the result back to the frontend return jsonify({ 'total_fuel_cost': total_fuel_cost, 'total_emissions': total_emissions })
